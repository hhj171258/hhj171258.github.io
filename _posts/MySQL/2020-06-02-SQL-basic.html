Creat table

	create table [table]
(
	[column1] [type] not null,
	[column2] [type],
	[column3] [type],
	primary key([colum1])
);

Drop table

	drop table (if exists) [table];

Select from

	Select * from [table]
where [column1] = '[value1]' and(or) [column2] like '[value2]%'
group by [column1]
order by [column1] asc, [column2], desc
limit ([number]), [number];
Select sum([column1]) (as ([alias])) from [table];
Select count([column1]) from [table];
Select avg([column1]) from [table];
Select distinct([column1]) from [table];
Select max([column1]) from [table];

Insert into

	insert into [table] (([column1], [column2])) values('[values1]','[values2]');

Update

	update [table]
set [column1] = '[value1]',
	[column2] = '[value2]'
where [colum3] = '[value3]';

Delete from

	delete from [table]
where [column1] = '[value1]';

Alter table

	alter table [table]
	modify [column] [type];
alter table [table]
	change [column] [columnNew] [type];
alter table [table]
	add [column] [type];
alter table [table]
	drop [column];
