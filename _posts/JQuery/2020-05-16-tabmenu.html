속성값 이용

	<ul class="tabmenu">
	<li data="tab1">content</li>
	<li data="tab2">content</li>
	<li data="tab3">content</li>
</ul>
<div class="tabcontent">
	<div id="tab1">content</div>
	<div id="tab2">content</div>
	<div id="tab3">content</div>
</div>
<script>
function tabmenu(tab, content){
	$(tab).first().addClass("on");
	$(content).hide().first().show();
	
	$(tab).click(function(){
		$(tab).removeClass("on");
		$(this).addClass("on");

		var tabData = $(this).attr("data");
		$(content).hide();
		$("#" + tabData).show();
	})
}
$(function(){
	tabmenu(".tabmenu li", ".tabcontent div");
})
</script>

서치해보면 대부분 이렇게 element에 속성을 넣고 값이 같으면 불러오는 방식을 이용한다.

.index() 이용

	<ul class="tabmenu">
	<li>content</li>
	<li>content</li>
	<li>content</li>
</ul>
<div class="tabcontent">content</div>
<div class="tabcontent">content</div>
<div class="tabcontent">content</div>
<script>
function tabmenu(tab, content){
	$(tab).eq(0).addClass("on");
	$(content).hide().eq(0).show();
	
	$(tab).click(function(){
		$(tab).removeClass("on");
		$(this).addClass("on");

		var contentIndex = $(this).index();
		$(content).hide();
		$(content).eq(contentIndex).show();
	})
}
$(function(){
	tabmenu(".tabmenu li", "tabcontent");
})
</script>

.index()를 이용하면 element에 속성을 넣지 않아도 될 것 같아서 만들었다. 나는 이 방법을 자주 사용한다.
