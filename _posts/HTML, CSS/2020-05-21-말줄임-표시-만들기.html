1줄

	.your-class{
	overflow: hidden;
	text-overflow: ellipsis;
	white-space: nowrap
}

2줄 이상

	.your-class{
	overflow: hidden;
	display: -webkit-box;
	-webkit-line-clamp: [number];
	-webkit-box-orient: vertical;
	height: [height];
	text-overflow: ellipsis;
}
2020-05-16

<input> placeholder

	:placeholder-shown{
	overflow: hidden;
	text-overflow: ellipsis;
	white-space: nowrap
}
:placeholder-shown{
	overflow: hidden;
	display: -webkit-box;
	-webkit-line-clamp: [number];
	-webkit-box-orient: vertical;
	height: [height];
	text-overflow: ellipsis;
}
