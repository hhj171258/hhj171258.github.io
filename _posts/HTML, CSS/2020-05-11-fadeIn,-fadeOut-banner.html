	<div class="wrap">
<div></div>
<div></div>
<div></div>				
</div>

	div:not(.wrap){position: absolute;left: 0;top: 0;width: 100%; animation: silde 18s infinite;}

div:nth-child(2){background-image: url("[url]"); animation-delay: 6s; opacity: 0;}
div:nth-child(3){background-image: url("[url]"); animation-delay: 12s; opacity: 0;}

	0% {opacity: 0;}
	11.11% {opacity: 1;}
	33.33% {opacity: 1;}
